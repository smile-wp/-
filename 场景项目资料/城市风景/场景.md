# 场景

## led和触摸

```c++
// 引脚定义
const int touchPin = 2;    // 触摸传感器连接数字引脚2
const int ledPin = 13;     // LED连接数字引脚13

// 变量定义
int touchState = 0;        // 当前触摸状态
int lastTouchState = 0;    // 上一次触摸状态
boolean ledState = false;  // LED当前状态

void setup() {
  Serial.begin(9600);
  pinMode(touchPin, INPUT);
  pinMode(ledPin, OUTPUT);
  
  // 初始状态LED熄灭
  digitalWrite(ledPin, LOW);
  
  Serial.println("触摸切换LED系统启动");
  Serial.println("每次触摸切换LED状态");
}

void loop() {
  // 读取触摸传感器状态
  touchState = digitalRead(touchPin);
  
  // 检测触摸信号的上升沿（从无触摸到有触摸）
  if (touchState == HIGH && lastTouchState == LOW) {
    // 切换LED状态
    ledState = !ledState;
    digitalWrite(ledPin, ledState);
    
    Serial.print("触摸检测 - LED状态: ");
    Serial.println(ledState ? "点亮" : "熄灭");
    
    delay(200); // 防抖延时
  }
  
  // 更新上一次触摸状态
  lastTouchState = touchState;
  
  delay(50);
}
```



## 加湿器

加湿器模块 → 继电器 → Arduino

加湿器 VCC → 继电器 COM
外部5V+ → 继电器 NO
外部GND → 加湿器 GND
继电器 IN → Arduino 引脚7
继电器 VCC → Arduino 5V
继电器 GND → Arduino GND

```c++
// 引脚定义
const int TOUCH_PIN = 2;
const int LED_PIN = 13;
const int HUMIDIFIER_PIN = 7;

// 状态变量
bool devicesRunning = false;    // 设备运行状态
int lastTouchState = LOW;
unsigned long touchTime = 0;
const unsigned long DEBOUNCE_DELAY = 300; // 防抖延时

void setup() {
  Serial.begin(9600);
  pinMode(TOUCH_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(HUMIDIFIER_PIN, OUTPUT);
  
  // 确保初始状态为关闭
  turnOffAllDevices();
  
  Serial.println("🔄 同步控制系统启动");
  Serial.println("✅ 触摸一次: LED和加湿器同时开启");
  Serial.println("✅ 再触摸一次: LED和加湿器同时关闭");
  Serial.println();
}

void loop() {
  int currentTouch = digitalRead(TOUCH_PIN);
  
  // 检测有效的触摸信号
  if (currentTouch == HIGH && lastTouchState == LOW) {
    if (millis() - touchTime > DEBOUNCE_DELAY) {
      // 切换设备状态
      devicesRunning = !devicesRunning;
      
      // 同步控制两个设备
      if (devicesRunning) {
        turnOnAllDevices();
        Serial.println("🚀 设备启动: LED亮 + 加湿器工作");
      } else {
        turnOffAllDevices();
        Serial.println("🛑 设备停止: LED灭 + 加湿器关闭");
      }
      
      touchTime = millis();
    }
  }
  
  lastTouchState = currentTouch;
  delay(50);
}

// 同时开启所有设备
void turnOnAllDevices() {
  digitalWrite(LED_PIN, HIGH);
  digitalWrite(HUMIDIFIER_PIN, LOW);
}

// 同时关闭所有设备
void turnOffAllDevices() {
  digitalWrite(LED_PIN, LOW);
  digitalWrite(HUMIDIFIER_PIN, HIGH);
}
```

## 加入音乐模块

```c++
#include "Arduino.h"
#include "SoftwareSerial.h"
#include "DFRobotDFPlayerMini.h"

SoftwareSerial mySoftwareSerial(10, 11);
DFRobotDFPlayerMini myDFPlayer;

const int TOUCH_PIN = 2;
const int LED_PIN = 13;
const int HUMIDIFIER_PIN = 7;

bool systemState = false;
int lastTouchState = LOW;

void setup() {
  Serial.begin(9600);
  
  // 先初始化基础设备
  pinMode(TOUCH_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(HUMIDIFIER_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);
  digitalWrite(HUMIDIFIER_PIN, LOW);
  
  Serial.println("基础设备初始化完成");
  
  // 尝试初始化音乐模块（非阻塞）
  delay(2000);
  mySoftwareSerial.begin(9600);
  
  Serial.println("尝试连接DFPlayer...");
  delay(2000);
  
  bool dfPlayerReady = false;
  
  // 尝试多次初始化
  for(int i = 0; i < 3; i++) {
    if (myDFPlayer.begin(mySoftwareSerial)) {
      dfPlayerReady = true;
      Serial.println("DFPlayer连接成功!");
      myDFPlayer.volume(30);
      myDFPlayer.EQ(0);
      delay(100);
      myDFPlayer.stop();
      break;
    } else {
      Serial.println("DFPlayer尝试连接失败，5秒后重试...");
      delay(5000);
    }
  }
  
  if (!dfPlayerReady) {
    Serial.println("警告: DFPlayer未就绪，系统将在无音乐模式下运行");
  }
  
  Serial.println("系统就绪 - 触摸控制LED和加湿器");
}

void loop() {
  int touchState = digitalRead(TOUCH_PIN);
  
  if (touchState == HIGH && lastTouchState == LOW) {
    systemState = !systemState;
    
    if (systemState) {
      // 开启所有设备
      digitalWrite(LED_PIN, HIGH);
      digitalWrite(HUMIDIFIER_PIN, LOW);
      
      // 尝试播放音乐（如果DFPlayer就绪）
      myDFPlayer.loop(1);
      Serial.println("系统启动 - 所有设备工作");
    } else {
      // 关闭所有设备
      digitalWrite(LED_PIN, LOW);
      digitalWrite(HUMIDIFIER_PIN, HIGH);
      myDFPlayer.stop();
      Serial.println("系统停止 - 所有设备关闭");
    }
    
    delay(500);
  }
  
  lastTouchState = touchState;
  delay(100);
}
```
