# 狮子跳火圈模块独立测试代码

根据您的要求，基于您提供的平滑移动示例，以下是狮子跳火圈模块的独立代码。此代码使用 MG90S 180度舵机，实现平滑的角度移动，速度和目标角度均可调节。

## 电路连接

- 舵机信号线 → Arduino 引脚 10（可根据实际修改）
- 舵机电源线 → 外部5V电源正极
- 舵机地线 → 外部5V电源负极 & Arduino GND

## 代码

```c
#include <Servo.h>

// 引脚定义
const int LION_SERVO_PIN = 10;  // 舵机信号引脚

Servo lionServo;                 // 创建舵机对象

// 可调节参数（在上传前根据需要修改）
int startAngle = 40;             // 起始角度（准备位置）
int endAngle = 140;              // 结束角度（跳过位置）
int speedDelay = 15;             // 每次转动延迟（毫秒），越大速度越慢（范围建议5~50）

// 状态变量
int currentAngle = 90;           // 当前角度（初始设为中间值）
int targetAngle = 90;            // 目标角度

void setup() {
  Serial.begin(9600);
  Serial.println("狮子跳火圈模块测试开始");

  lionServo.attach(LION_SERVO_PIN);
  lionServo.write(currentAngle); // 初始化到中间位置
  delay(1000);
}

void loop() {
  // 演示自动跳跃循环：从起始角度到结束角度往复运动
  smoothMove(endAngle);          // 跳起到结束角度
  delay(1000);                   // 保持1秒（模拟跳过火圈后的停留）

  smoothMove(startAngle);        // 返回起始角度
  delay(1000);                   // 保持1秒

  // 可选：通过串口实时调整参数（解除注释下一行即可启用）
  // handleSerialCommands();
}

// 平滑移动函数（与示例一致，但使用全局变量）
void smoothMove(int target) {
  targetAngle = target;

  while (currentAngle != targetAngle) {
    if (currentAngle < targetAngle) {
      currentAngle++;
    } else {
      currentAngle--;
    }
    lionServo.write(currentAngle);
    delay(speedDelay);            // 控制速度
  }

  // 移动完成后打印当前角度（用于调试）
  Serial.print("到达角度: ");
  Serial.println(currentAngle);
}

// 可选：通过串口命令动态调整参数（如需启用，请在loop中调用）
void handleSerialCommands() {
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if (cmd.startsWith("start ")) {
      startAngle = cmd.substring(6).toInt();
      startAngle = constrain(startAngle, 0, 180);
      Serial.print("起始角度设为: ");
      Serial.println(startAngle);
    }
    else if (cmd.startsWith("end ")) {
      endAngle = cmd.substring(4).toInt();
      endAngle = constrain(endAngle, 0, 180);
      Serial.print("结束角度设为: ");
      Serial.println(endAngle);
    }
    else if (cmd.startsWith("speed ")) {
      speedDelay = cmd.substring(6).toInt();
      speedDelay = constrain(speedDelay, 2, 100);
      Serial.print("速度延迟设为: ");
      Serial.println(speedDelay);
    }
    else if (cmd == "jump") {
      smoothMove(endAngle);
      delay(500);
      smoothMove(startAngle);
    }
    else if (cmd == "help") {
      Serial.println("可用命令: start [角度], end [角度], speed [延迟], jump, help");
    }
  }
}
```

## 使用说明

1. **基本测试**
   上传代码后，舵机将自动在 `startAngle` 和 `endAngle` 之间来回运动，每步延迟 `speedDelay` 毫秒，到达后停留1秒。
2. **调整参数**
   - 修改 `startAngle`、`endAngle` 和 `speedDelay` 的初始值（代码开头部分），然后重新上传。
   - 角度范围：0~180度（请根据实际机械结构调整，避免超出舵机行程）。
   - 速度：`speedDelay` 越小，运动越快；建议范围 5~50。
3. **串口实时控制（高级）**
   如需通过串口动态调整参数，请在 `loop()` 中取消注释 `handleSerialCommands();`，然后通过串口监视器发送以下命令（波特率9600）：
   - `start 40` → 设置起始角度为40
   - `end 140` → 设置结束角度为140
   - `speed 20` → 设置速度延迟为20
   - `jump` → 立即执行一次跳跃（到结束角度再返回起始）
   - `help` → 显示帮助信息
4. **注意事项**
   - 确保舵机电源足够（外部5V供电，不要直接从Arduino 5V引脚取电）。
   - 首次运行请观察舵机运动范围，避免机械卡死。

#include <Servo.h>

// ===== 引脚定义 =====
const int PIN_LION_SERVO = 10;   // 狮子舵机 (180度)
const int PIN_GEAR_SERVO = 11;   // 齿轮舵机 (360度)

int lionTargetAngle = 90;        // 目标角度

int lionEndAngle = 140;          // 结束角度（跳过）
int lionSpeedDelay = 15;         // 步进间隔（毫秒），值越大越慢
bool lionIsJumping = false;      // 是否正在跳跃中
unsigned long lionLastStepTime = 0; // 上次步进时间

// ===== 齿轮模块参数 =====
// 360度连续旋转舵机速度说明：
// 90 = 停止
// 0~89 = 顺时针旋转（0最快，89最慢）
// 91~180 = 逆时针旋转（91最慢，180最快）
const int GEAR_SPEED = 89;       // 顺时针中等速度（可根据需要修改）

// 舵机对象
Servo lionServo;
Servo gearServo;

void setup() {
  Serial.begin(9600);
  Serial.println("狮子+齿轮模块启动（参考示例代码）");

  // 舵机连接
  lionServo.attach(PIN_LION_SERVO);
  gearServo.attach(PIN_GEAR_SERVO);

  // 初始化狮子位置
  lionServo.write(lionCurrentAngle);

  // 设置齿轮持续旋转（写入一次即可）
  gearServo.write(GEAR_SPEED);
  Serial.print("齿轮旋转速度设置为: ");
  Serial.println(GEAR_SPEED);
  Serial.println("齿轮开始持续旋转（90停止，<90顺时针，>90逆时针）");
}

void loop() {
  // 处理狮子模块的非阻塞移动
  updateLionJump();

  // 可选：串口命令控制狮子（需取消注释下方函数）
  // handleSerialCommands();
}

/**
 * 狮子跳火圈非阻塞更新函数
 * 在loop中频繁调用，根据时间步进角度
 */
void updateLionJump() {
  if (!lionIsJumping) {
    // 不在跳跃中，可触发自动跳跃（例如每3秒跳一次）
    static unsigned long lastAutoJump = 0;
    if (millis() - lastAutoJump > 3000) { // 自动间隔3秒
      lastAutoJump = millis();
      startLionJump();
    }
    return;
  }

  // 正在跳跃，判断是否需要步进
  if (millis() - lionLastStepTime >= lionSpeedDelay) {
    lionLastStepTime = millis();

    if (lionCurrentAngle < lionTargetAngle) {
      lionCurrentAngle++;
    } else if (lionCurrentAngle > lionTargetAngle) {
      lionCurrentAngle--;
    }

    lionServo.write(lionCurrentAngle);

    // 检查是否到达目标
    if (lionCurrentAngle == lionTargetAngle) {
      if (lionTargetAngle == lionEndAngle) {
        // 到达最高点，准备返回
        lionTargetAngle = lionStartAngle;
        Serial.println("狮子跳过火圈！");
      } else {
        // 返回起点，跳跃结束
        lionIsJumping = false;
        Serial.println("狮子回到起始位置");
      }
    }
  }
}

/**
 * 触发一次跳跃
 */
void startLionJump() {
  if (!lionIsJumping) {
    lionIsJumping = true;
    lionTargetAngle = lionEndAngle;
    Serial.println("狮子开始跳火圈！");
  }
}

/**
 * 设置狮子参数（起始角度、结束角度、速度等级）
 * 速度等级 1~100 映射到步进延迟 50~5 毫秒
 */
void setLionParams(int start, int end, int speed) {
  lionStartAngle = constrain(start, 0, 180);
  lionEndAngle = constrain(end, 0, 180);
  // speed: 1~100 映射到延迟 50~5 ms
  lionSpeedDelay = map(speed, 1, 100, 50, 5);
  Serial.print("狮子参数更新: 起始=");
  Serial.print(lionStartAngle);
  Serial.print(" 结束=");
  Serial.print(lionEndAngle);
  Serial.print(" 速度=");
  Serial.println(speed);
}

/**
 * 串口命令处理（可选）
 * 发送命令格式：
 *   start 40     设置起始角度
 *   end 140      设置结束角度
 *   speed 30     设置速度（1-100）
 *   jump         立即跳跃一次
 *   help         显示帮助
 */

  if (!Serial.available()) return;

  String cmd = Serial.readStringUntil('\n');
  cmd.trim();

  if (cmd.startsWith("start ")) {
    int val = cmd.substring(6).toInt();
    lionStartAngle = constrain(val, 0, 180);
    Serial.print("起始角度=");
    Serial.println(lionStartAngle);
  }
  else if (cmd.startsWith("end ")) {
    int val = cmd.substring(4).toInt();
    lionEndAngle = constrain(val, 0, 180);
    Serial.print("结束角度=");
    Serial.println(lionEndAngle);
  }
  else if (cmd.startsWith("speed ")) {
    int val = cmd.substring(6).toInt();
    val = constrain(val, 1, 100);
    lionSpeedDelay = map(val, 1, 100, 50, 5);
    Serial.print("速度延迟=");
    Serial.println(lionSpeedDelay);
  }
  else if (cmd == "jump") {
    startLionJump();
  }
  else if (cmd == "help") {
    Serial.println("可用命令: start [角度], end [角度], speed [1-100], jump, help");
  }
}

        Serial.println("狮子跳过火圈！");
      } else {
        lionIsJumping = false;
        Serial.println("狮子回到起始位置");
      }
    }
  }
}

// ==================== 升降模块函数 ====================
void updateLiftModule(unsigned long now) {
  // 自动模式：定期切换目标位置（最低<->最高）
  if (liftAutoMode && !liftIsMoving) {
    if (now - liftLastAutoToggle >= LIFT_AUTO_INTERVAL) {
      liftLastAutoToggle = now;
      // 如果当前在最低或接近最低，就前往最高；否则前往最低
      if (liftCurrentAngle <= (liftMinAngle + liftMaxAngle) / 2) {
        liftTargetAngle = liftMaxAngle;
      } else {
        liftTargetAngle = liftMinAngle;
      }
      liftIsMoving = true;
      Serial.print("升降自动切换目标: ");
      Serial.println(liftTargetAngle);
    }
  }

  // 非阻塞移动
  if (liftIsMoving && (now - liftLastStepTime >= liftSpeedDelay)) {
    liftLastStepTime = now;

    if (liftCurrentAngle < liftTargetAngle) {
      liftCurrentAngle++;
    } else if (liftCurrentAngle > liftTargetAngle) {
      liftCurrentAngle--;
    }

    liftServo.write(liftCurrentAngle);

    // 到达目标
    if (liftCurrentAngle == liftTargetAngle) {
      liftIsMoving = false;
      Serial.print("升降到达目标位置: ");
      Serial.println(liftCurrentAngle);
    }
  }
}

// ==================== 串口命令处理（可选）====================
void handleLiftCommands() {
  if (!Serial.available()) return;

  String cmd = Serial.readStringUntil('\n');
  cmd.trim();

  if (cmd.startsWith("lift_to ")) {
    int val = cmd.substring(7).toInt();
    setLiftTarget(val);
  }
  else if (cmd.startsWith("lift_speed ")) {
    int val = cmd.substring(11).toInt();
    setLiftSpeed(val);
  }
  else if (cmd == "lift_auto") {
    toggleLiftAuto();
  }
  else if (cmd == "lift_help") {
    Serial.println("升降命令: lift_to [角度], lift_speed [1-100], lift_auto");
  }
}

  else if (cmd == "jump") {

  }
  else if (cmd == "help") {
    Serial.println("可用命令: start [角度], end [角度], speed [1-100], jump, help");
  }
}
