## 1.触摸传感器控制SG90舵机（180）转动系统

要实现触摸传感器控制SG90舵机转动的系统，关键在于搭建合理的硬件电路并编写控制逻辑程序。以下是分步骤实现方案，整合了触摸检测、信号处理和舵机驱动的完整流程：

------

### 🛠 一、硬件选型与连接

1. **核心控制器**

   - **推荐Arduino UNO**：易用性强，自带PWM输出和数字输入接口，支持`Servo`库10。
   - 替代方案：树莓派（需Python编程）3 或51单片机（需手动生成PWM）9。

2. **模块接线**

   - **触摸传感器**：
     - `VCC` → 5V电源
     - `GND` → 控制器GND
     - `SIG` → 控制器数字引脚（如Arduino D2）10
   - **SG90舵机**：
     - `橙色/黄色线（SIG）` → 控制器PWM引脚（如Arduino D9）
     - `红色线（VCC）` → 5V电源（⚠️若舵机≥3个需外接2A电源）
     - `棕色/黑色线（GND）` → 控制器GND（必须与触摸传感器共地）710

   plaintext

   ```
   ⚡电源注意事项：
   - 单舵机：可直接用Arduino的5V引脚供电（电流≤500mA）。
   - 多舵机：外接5V/2A稳压电源，避免控制器重启:cite[10]。
   ```

------

### ⚙️ 二、控制逻辑设计

1. **触摸检测**
   - 触摸传感器输出高电平（触摸时）或低电平（空闲时）。
   - 代码中需加入**消抖处理**（延时20~50ms），避免误触发9。
2. **舵机角度控制**
   - SG90的PWM参数：
     - 周期：20ms（50Hz）
     - 脉宽范围：0.5ms（0°） ~ 2.5ms（180°）38
   - 在Arduino中，直接用`Servo.write(angle)`设置角度（0~180）10。



### ⚙️ 三、代码实现（arduino）

```
#include <Servo.h>

const int touchPin = 2;    // 触摸传感器接D2
const int servoPin = 9;    // 舵机接D9
Servo myServo;             // 创建舵机对象

void setup() {
  pinMode(touchPin, INPUT);
  myServo.attach(servoPin); // 初始化舵机
  myServo.write(90);        // 初始位置设为90°
}

void loop() {
  if (digitalRead(touchPin) == HIGH) { // 检测触摸
    delay(30);                        // 消抖处理
    myServo.write(0);                 // 转到0°位置
    delay(1000);                      // 保持1秒
    myServo.write(180);               // 转到180°位置
    delay(1000);                      
    myServo.write(90);                // 返回中心位置
  }
}
```

## 2. 触摸传感器控制SG90舵机（360）转动系统

设计一个完整的触摸旋转台系统，使用触摸传感器和SG90 360°舵机实现"触摸旋转、松开停止"的功能。下面是完整的解决方案：

### 一、🔄 SG90 360°舵机特性

| 控制信号  | 舵机行为                       |
| :-------- | :----------------------------- |
| 1.5ms脉冲 | **完全停止**                   |
| 1.3ms脉冲 | **全速逆时针旋转**             |
| 1.7ms脉冲 | **全速顺时针旋转**             |
| 1.0-1.5ms | 逆时针旋转（脉宽越小转速越快） |
| 1.5-2.0ms | 顺时针旋转（脉宽越大转速越快） |

> 📌 注意：360°舵机**无法控制精确角度**，只能控制旋转方向和速度

### 二、系统设计思路

- **触摸传感器**：作为旋转台的启动开关
- **SG90 360°舵机**：提供连续旋转动力
- **实时响应**：触摸即转，松开即停
- **安全保护**：防止过热和过载
- **视觉反馈**：LED指示灯显示系统状态

```c++
#include <Servo.h>

// 引脚定义
const int TOUCH_PIN = 2;      // 触摸传感器引脚
const int SERVO_PIN = 9;      // 舵机控制引脚
const int LED_PIN = 13;       // 状态指示灯引脚

// 舵机控制参数
const int STOP_PULSE = 1500;  // 停止脉冲 (1500μs)
const int ROTATE_PULSE = 1700; // 顺时针旋转脉冲 (1700μs)
// 如需逆时针旋转: const int ROTATE_PULSE = 1300;

Servo rotationServo;          // 舵机对象

// 系统状态变量
bool isRotating = false;      // 当前旋转状态
unsigned long rotateStartTime = 0; // 旋转开始时间
const unsigned long MAX_CONTINUOUS_TIME = 300000; // 最大连续工作时间5分钟

void setup() {
  // 初始化串口通信
  Serial.begin(9600);
  Serial.println("旋转台系统启动...");
  
  // 设置引脚模式
  pinMode(TOUCH_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  
  // 初始化舵机
  rotationServo.attach(SERVO_PIN);
  
  // 初始校准 - 确保舵机停止
  rotationServo.writeMicroseconds(STOP_PULSE);
  digitalWrite(LED_PIN, LOW); // LED熄灭表示停止
  
  Serial.println("系统初始化完成，等待触摸...");
}

void loop() {
  // 检测触摸状态
  bool touchState = digitalRead(TOUCH_PIN);
  
  // 触摸状态发生变化
  if (touchState == HIGH && !isRotating) {
    // 开始旋转
    startRotation();
  } else if (touchState == LOW && isRotating) {
    // 停止旋转
    stopRotation();
  }
  
  // 安全保护：检查连续工作时间
  if (isRotating && (millis() - rotateStartTime > MAX_CONTINUOUS_TIME)) {
    Serial.println("安全保护：连续工作时间超过5分钟，自动停止！");
    stopRotation();
    digitalWrite(LED_PIN, HIGH);
    delay(1000);
    digitalWrite(LED_PIN, LOW);
    delay(1000);
  }
  
  // 短延时保持系统稳定
  delay(10);
}

// 开始旋转函数
void startRotation() {
  rotationServo.writeMicroseconds(ROTATE_PULSE);
  digitalWrite(LED_PIN, HIGH); // LED亮表示旋转中
  isRotating = true;
  rotateStartTime = millis();
  Serial.println("旋转启动");
}

// 停止旋转函数
void stopRotation() {
  rotationServo.writeMicroseconds(STOP_PULSE);
  digitalWrite(LED_PIN, LOW); // LED灭表示停止
  isRotating = false;
  
  // 计算本次旋转时长
  unsigned long duration = millis() - rotateStartTime;
  Serial.print("旋转停止，本次工作时长：");
  Serial.print(duration / 1000);
  Serial.println("秒");
  
  // 安全冷却时间（每工作1分钟冷却10秒）
  if (duration > 60000) {
    unsigned long cooldown = duration / 6000; // 每工作1分钟冷却10秒
    Serial.print("系统冷却中，请等待 ");
    Serial.print(cooldown);
    Serial.println(" 秒...");
    delay(cooldown * 1000);
  }
}
```

## 3.压力传感器控制SG90舵机（360）转动系统

SG90 360度舵机与普通180度舵机不同，它通过PWM信号控制的是旋转方向和速度，而不是固定角度。以下是专门针对SG90 360度舵机的控制方案。

### 一、硬件连接

- 压力传感器 → Arduino模拟输入引脚(A0)
- SG90 360度舵机 → Arduino数字PWM引脚(9)
- 共地连接

### 二、控制原理

对于360度舵机：

- **90°信号(1.5ms脉冲)**：停止转动
- **>90°信号(如180°, 2ms脉冲)**：全速顺时针旋转
- **<90°信号(如0°, 1ms脉冲)**：全速逆时针旋转
- 中间值控制旋转速度

### 三、完整代码实现

arduino

```c
#include <Servo.h>

Servo myServo;  // 创建舵机对象

const int pressurePin = A0;  // 压力传感器连接引脚
const int servoPin = 9;      // 舵机连接引脚
const int threshold = 50;    // 压力阈值，根据实际传感器调整

int pressureValue = 0;       // 存储压力值
int servoSignal = 90;        // 90为停止信号
bool isMoving = false;       // 标记舵机是否在转动

void setup() {
  myServo.attach(servoPin);  // 将舵机连接到指定引脚
  Serial.begin(9600);        // 初始化串口通信
  myServo.write(90);         // 初始状态为停止
}

void loop() {
  // 读取压力传感器值（0-1023）
  pressureValue = analogRead(pressurePin);
  
  if (pressureValue > threshold) {
    // 有压力时，根据压力值计算舵机信号
    // 映射到70-110范围(可根据需要调整)
    servoSignal = map(pressureValue, threshold, 1023, 70, 110);
    
    // 限制在安全范围内(通常60-120)
    servoSignal = constrain(servoSignal, 60, 120);
    
    myServo.write(servoSignal);
    isMoving = true;
    
    Serial.print("Pressure detected: ");
    Serial.print(pressureValue);
    Serial.print("\t Servo Signal: ");
    Serial.println(servoSignal);
  } 
  else {
    // 无压力时，舵机停止
    if (isMoving) {
      myServo.write(90);  // 发送停止信号
      isMoving = false;
      Serial.println("No pressure - servo stopped");
    }
  }
  
  delay(20);  // 短暂延迟稳定读取
}
```

参数调整指南

1. **旋转方向控制**：

   - `map(pressureValue, threshold, 1023, 70, 110)`中的70-110范围：
     - 70-89：逆时针旋转(值越小速度越快)
     - 91-110：顺时针旋转(值越大速度越快)
     - 90：停止

2. **速度控制**：

   - 调整映射范围(如60-120)可增加最大速度
   - 但不要超出建议的60-120范围，以免损坏舵机

3. **死区设置**：

   arduino

   ```
   // 在90附近设置死区，避免微小抖动
   if (servoSignal > 85 && servoSignal < 95) {
     servoSignal = 90;
   }
   ```

进阶功能实现

1. **压力敏感速度控制**：

```
  // 正压力顺时针，负压力逆时针(如果使用双方向压力传感器)
```

### 四、注意事项

1. SG90 360度舵机不能精确定位，只能控制旋转方向和速度
2. 长时间全速运转可能导致过热，建议间歇工作
3. 实际PWM范围可能因舵机个体差异而不同，需要测试调整
4. 确保电源充足(5V/1A以上)，最好舵机单独供电
5. 首次测试时建议用手轻轻握住舵机，观察旋转方向是否正确

这个方案实现了压力传感控制360度舵机的基本功能：有压力时按压力大小控制旋转方向和速度，无压力时完全停止。您可以根据具体应用需求调整控制参数。

```
const int STOP_PULSE = 1500;     // 停止脉冲 (1500μs)
const int ROTATE_PULSE = 1700;   // 顺时针旋转脉冲 (1700μs)
// 如需逆时针旋转: const int ROTATE_PULSE = 1300;

Servo rotationServo;             // 舵机对象

// 磁场检测参数
const int MAGNET_THRESHOLD = 500; // 磁场触发阈值 (根据实际调整)
const int HYSTERESIS = 50;       // 迟滞值防止抖动
bool isRotating = false;         // 当前旋转状态

void setup() {
  // 初始化串口通信
  Serial.begin(9600);
  Serial.println("磁控旋转系统启动...");
  
  // 初始化舵机
  rotationServo.attach(SERVO_PIN);
  rotationServo.writeMicroseconds(STOP_PULSE);
  
  Serial.println("系统初始化完成，等待磁场...");
  Serial.println("--------------------------------");
  Serial.println("磁场强度 | 状态");
  Serial.println("--------------------------------");
}

void loop() {
  // 读取霍尔传感器值 (0-1023)
  int hallValue = analogRead(HALL_SENSOR_PIN);
  
  // 串口输出磁场强度
  Serial.print(hallValue);
  Serial.print(" | ");
  
  // 磁铁靠近检测 (带迟滞滤波)
  if (!isRotating && hallValue > (MAGNET_THRESHOLD + HYSTERESIS)) {
    startRotation();
    isRotating = true;
  }
  // 磁铁远离检测
  else if (isRotating && hallValue < (MAGNET_THRESHOLD - HYSTERESIS)) {
    stopRotation();
    isRotating = false;
  }
  
  // 短延时保持系统稳定
  delay(100);
}

// 开始旋转函数
void startRotation() {
  rotationServo.writeMicroseconds(STOP_PULSE);
  Serial.println("旋转启动");
}

// 停止旋转函数
void stopRotation() {
  rotationServo.writeMicroseconds(ROTATE_PULSE);
  Serial.println("旋转停止");
}
```

```
const unsigned long STOP_DELAY = 2000; // 磁铁远离后2秒停止

void loop() {
  int hallValue = analogRead(HALL_SENSOR_PIN);
  static unsigned long lastDetection = 0;
  
  if (hallValue > MAGNET_THRESHOLD) {
    rotationServo.writeMicroseconds(ROTATE_PULSE);
    lastDetection = millis();
  }
  else if (millis() - lastDetection > STOP_DELAY) {
    rotationServo.writeMicroseconds(STOP_PULSE);
  }
  delay(50);
}
```
