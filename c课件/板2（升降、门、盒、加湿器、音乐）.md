#include <Servo.h>

// 定义霍尔传感器引脚
const int HALL_LIFT1_PIN = 2;      // 升降控制1
const int HALL_LIFT2_PIN = 3;      // 升降控制2
const int HALL_DOOR_PIN = 7;       // 门控制
const int HALL_BOX_PIN = 5;        // 盒子开合控制
const int HALL_HUMID_PIN = 6;      // 加湿器控制

// 定义执行器引脚
const int SERVO_LIFT1_PIN = 8;     // 升降舵机1
const int SERVO_LIFT2_PIN = 9;     // 升降舵机2

const int SERVO_BOX_PIN = 11;      // 盒子开合舵机
const int RELAY_HUMID_PIN = 12;    // 加湿器继电器

// 创建软件串口对象和DFPlayer对象
SoftwareSerial mySoftwareSerial(A0, A1); // RX, TX (使用模拟引脚作为数字引脚)

// 创建舵机对象
Servo liftServo1;
Servo liftServo2;
Servo doorServo;
Servo boxServo;

// 状态变量
int liftAngle1 = 0;                // 升降舵机1角度
int liftAngle2 = 0;                // 升降舵机2角度

int boxAngle = 0;                  // 盒子舵机角度
bool humidifierState = false;      // 加湿器状态

// 升降模块1专用变量
bool lastLift1MagnetState = false; // 上一次磁铁状态(升降1)
bool lift1ToggleState = false;     // 升降1切换状态

// 升降模块2专用变量
bool lastLift2MagnetState = false; // 上一次磁铁状态(升降2)
bool lift2ToggleState = false;     // 升降2切换状态

// 门控制模块专用变量
bool lastDoorMagnetState = false;  // 上一次磁铁状态(门)

// 盒子控制模块专用变量

unsigned long lastLift1DebounceTime = 0;

unsigned long lastHumidDebounceTime = 0;
const unsigned long debounceDelay = 300; // 防抖延迟(ms)

// 舵机状态变量 - 新增
bool lift1Active = false;
bool lift2Active = false;
bool doorActive = false;
bool boxActive = false;
unsigned long lift1LastActionTime = 0;
unsigned long lift2LastActionTime = 0;
unsigned long doorLastActionTime = 0;
unsigned long boxLastActionTime = 0;
const unsigned long servoDetachDelay = 1000; // 舵机动作后延迟1秒断开

// 传感器读取函数 - 增加数字滤波
bool readSensor(int pin) {
  // 读取多次并取平均值，减少干扰
  int count = 0;
  for (int i = 0; i < 5; i++) {
    if (digitalRead(pin) == LOW) count++;
    delay(1);
  }
  return (count >= 3); // 如果5次中有3次以上检测到磁铁，则认为有磁铁
}

// 舵机控制函数 - 新增
void controlServo(Servo &servo, int pin, int angle, bool &active, unsigned long &lastActionTime) {
  if (!active) {
    servo.attach(pin);
    active = true;
  }
  servo.write(angle);
  lastActionTime = millis();
}

// 检查并断开不活动的舵机 - 新增
void checkServoDetach() {
  unsigned long currentTime = millis();
  
  if (lift1Active && currentTime - lift1LastActionTime > servoDetachDelay) {
    liftServo1.detach();
    lift1Active = false;
    Serial.println("升降舵机1已断开");
  }
  
  if (lift2Active && currentTime - lift2LastActionTime > servoDetachDelay) {
    liftServo2.detach();
    lift2Active = false;
    Serial.println("升降舵机2已断开");
  }
  
  if (doorActive && currentTime - doorLastActionTime > servoDetachDelay) {
    doorServo.detach();
    doorActive = false;
    Serial.println("门舵机已断开");
  }
  
  if (boxActive && currentTime - boxLastActionTime > servoDetachDelay) {
    boxServo.detach();
    boxActive = false;
    Serial.println("盒子舵机已断开");
  }
}

void setup() {
  // 初始化串口
  Serial.begin(9600);
  Serial.println("系统启动中...");
  
  // 设置霍尔传感器引脚为输入模式并启用内部上拉电阻
  pinMode(HALL_LIFT1_PIN, INPUT_PULLUP);
  pinMode(HALL_LIFT2_PIN, INPUT_PULLUP);
  pinMode(HALL_DOOR_PIN, INPUT_PULLUP);
  pinMode(HALL_BOX_PIN, INPUT_PULLUP);
  pinMode(HALL_HUMID_PIN, INPUT_PULLUP);
  
  // 设置执行器引脚为输出模式
  pinMode(RELAY_HUMID_PIN, OUTPUT);
  
  // 测试所有舵机
  testServos();
  
  // 初始状态
  Serial.println("设置初始状态...");
  digitalWrite(RELAY_HUMID_PIN, LOW);
  
  // 初始化磁铁状态
  lastLift1MagnetState = readSensor(HALL_LIFT1_PIN);
  lastLift2MagnetState = readSensor(HALL_LIFT2_PIN);
  lastDoorMagnetState = readSensor(HALL_DOOR_PIN);
  lastBoxMagnetState = readSensor(HALL_BOX_PIN);
  
  // 初始化MP3播放器
  Serial.println("初始化MP3播放器...");
  mySoftwareSerial.begin(9600);
  
  if (!myDFPlayer.begin(mySoftwareSerial)) {
    Serial.println("无法初始化DFPlayer Mini!");
    Serial.println("1. 请检查接线是否正确");
    Serial.println("2. 请检查TF卡是否正确格式化");
    Serial.println("3. 请检查MP3文件夹和0003.mp3文件是否存在");
    // 不要停止程序，继续运行其他功能
  } else {
    Serial.println("DFPlayer Mini初始化成功");
    myDFPlayer.volume(20);   // 音量设定为20
    myDFPlayer.loop(3);      // 循环播放TF卡中0003.mp3文件
  }
  
  Serial.println("系统初始化完成");
  Serial.println("音乐开始自动播放");
  Serial.println("等待霍尔传感器触发...");
  
  // 打印引脚分配信息
  Serial.println("=== 引脚分配信息 ===");
  Serial.println("升降1霍尔传感器: " + String(HALL_LIFT1_PIN));
  Serial.println("升降2霍尔传感器: " + String(HALL_LIFT2_PIN));
  Serial.println("门霍尔传感器: " + String(HALL_DOOR_PIN));
  Serial.println("盒子霍尔传感器: " + String(HALL_BOX_PIN));

  Serial.println("升降舵机2: " + String(SERVO_LIFT2_PIN));

  Serial.println("盒子舵机: " + String(SERVO_BOX_PIN));
  Serial.println("加湿器继电器: " + String(RELAY_HUMID_PIN));
  Serial.println("MP3 RX: A0, TX: A1");
  Serial.println("====================");
}

void loop() {
  // 读取所有霍尔传感器状态并打印（用于调试）
  static unsigned long lastSensorReadTime = 0;
  if (millis() - lastSensorReadTime > 2000) {
    lastSensorReadTime = millis();
    Serial.print("传感器状态 - 升降1:");
    Serial.print(readSensor(HALL_LIFT1_PIN));
    Serial.print(" 升降2:");
    Serial.print(readSensor(HALL_LIFT2_PIN));
    Serial.print(" 门:");
    Serial.print(readSensor(HALL_DOOR_PIN));
    Serial.print(" 盒子:");
    Serial.print(readSensor(HALL_BOX_PIN));
    Serial.print(" 加湿器:");
    Serial.println(readSensor(HALL_HUMID_PIN));
  }
  
  // 处理升降控制1
  handleLift1Control();
  
  // 处理升降控制2
  handleLift2Control();
  
  // 处理门控制
  handleDoorControl();
  
  // 处理盒子控制
  handleBoxControl();
  
  // 处理加湿器控制
  handleHumidifierControl();
  
  // 检查并断开不活动的舵机 - 新增
  checkServoDetach();
  
  // 短暂延迟防止过于频繁的检测
  delay(50);
}

// 升降控制1函数
void handleLift1Control() {
  bool currentMagnetState = readSensor(HALL_LIFT1_PIN);
  
  // 检测磁铁状态变化 (从远离到靠近)
  if (currentMagnetState && !lastLift1MagnetState) {
    // 防抖检查
    if (millis() - lastLift1DebounceTime > debounceDelay) {
      lastLift1DebounceTime = millis();
      
      // 切换舵机状态
      if (lift1ToggleState) {
        liftAngle1 = 180;
        Serial.println("升降模块1: 旋转到180度");
      } else {
        liftAngle1 = 0;
        Serial.println("升降模块1: 旋转到0度");
      }
      
      // 更新舵机位置
      controlServo(liftServo1, SERVO_LIFT1_PIN, liftAngle1, lift1Active, lift1LastActionTime);
      
      // 切换下次状态
      lift1ToggleState = !lift1ToggleState;
    }
  }
  
  // 更新上一次磁铁状态
  lastLift1MagnetState = currentMagnetState;
}

// 加湿器控制函数
void handleHumidifierControl() {
  bool currentMagnetState = readSensor(HALL_HUMID_PIN);
  static bool lastMagnetState = false;
  
  // 检测磁铁状态变化 (从远离到靠近)
  if (currentMagnetState && !lastMagnetState) {
    // 防抖检查
    if (millis() - lastHumidDebounceTime > debounceDelay) {
      lastHumidDebounceTime = millis();
      
      humidifierState = !humidifierState;
      digitalWrite(RELAY_HUMID_PIN, humidifierState ? HIGH : LOW);
      
      Serial.print("加湿器: ");
      Serial.println(humidifierState ? "开启" : "关闭");
    }
  }
  
  // 更新上一次磁铁状态
  lastMagnetState = currentMagnetState;
}
