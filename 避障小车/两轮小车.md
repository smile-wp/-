# 两轮小车

# **智能小车冬令营课程设计**

## **主题：Arduino智能避障小车（L9110s版）**

## **课程概述**

本课程为期10天，将带领学生从零开始构建一辆具有智能避障功能的Arduino小车。课程融合了机械结构设计、3D建模与打印、激光切割、电子电路和编程控制等多学科知识，培养学生的工程思维和创新能力。

## **课程总览**

- **适合年龄**：10-16岁
- **课程时长**：每天2小时，共10天
- **学生人数**：每班8-12人（建议配2名助教）
- **教学工具**：Arduino IDE、Fusion 360、激光切割软件
- **成品**：每位学生完成一辆个性化智能避障小车

## **完整代码与详细注释**

cpp

```
/*
 * ARDUINO智能避障小车 - 冬令营教学版
 * 使用L9110s电机驱动板 + 双TT电机配置
 * 版本：1.0
 * 作者：冬令营教学团队
 * 日期：2024年1月
 * 
 * 课程适用：10-16岁学生
 * 教学目标：
 * 1. 理解超声波测距原理
 * 2. 掌握电机PWM控制
 * 3. 学习避障算法设计
 * 4. 培养工程调试能力
 */

/**************************************************
 *            第一部分：头文件与库引入
 * 教学提示：库就像是给Arduino的"超能力卡片"
 * 有了这些卡片，Arduino就能做更多事情
 **************************************************/
#include <NewPing.h>    // 超声波测距库 - 让距离测量更简单
#include <Servo.h>      // 舵机控制库 - 控制"头部"转动

/**************************************************
 *            第二部分：引脚定义
 * 教学提示：这里是小车的"神经系统"
 * 每个引脚就像神经末梢，连接不同器官
 * 
 * 引脚分配原则：
 * 1. 数字引脚：控制开关信号（电机、舵机）
 * 2. 模拟引脚：读取连续信号（本项目中未使用）
 * 3. 电源引脚：5V和GND供电
 **************************************************/
// 超声波传感器引脚（根据PDF第9页）
#define TRIG_PIN 6      // 触发引脚 - 发出"滴"声
#define ECHO_PIN 7      // 回波引脚 - 听到"回声"
#define MAX_DISTANCE 200 // 最大测量距离（厘米）

// L9110s电机驱动引脚（根据PDF第8页）
// 电机A（左轮）控制
#define MOTOR_A_PIN1 2   // MA方向控制1
#define MOTOR_A_PIN2 3   // MA方向控制2
// 电机B（右轮）控制
#define MOTOR_B_PIN1 4   // MB方向控制1
#define MOTOR_B_PIN2 5   // MB方向控制2

// 教学提示：为什么用这些引脚？
// D2-D5：支持PWM输出，可以控制电机速度
// D6,D7：数字IO，适合超声波传感器
// D9：PWM引脚，适合舵机控制

/**************************************************
 *            第三部分：速度参数设置
 * 教学提示：这些是小车的"性格参数"
 * 就像给小车设定走路快慢、反应速度
 * 
 * PWM值范围：0-255
 * 0 = 停止，255 = 全速
 **************************************************/
#define MAX_SPEED 200      // 最大速度（紧急情况使用）
#define CRUISE_SPEED 180   // 巡航速度（正常前进）
#define TURN_SPEED 150     // 转弯速度
#define SLOW_SPEED 100     // 慢速（接近障碍物时）

/**************************************************
 *            第四部分：避障参数设置
 * 教学提示：这些是小车的"视力参数"
 * 决定了小车能看到多远，多敏感
 **************************************************/
#define OBSTACLE_DISTANCE 20  // 障碍物检测距离（厘米）
#define SAFE_DISTANCE 30      // 安全距离（选择转弯方向）
#define CRITICAL_DISTANCE 10  // 紧急停止距离

/**************************************************
 *            第五部分：对象创建
 * 教学提示：创建小车的"器官对象"
 * 超声波是"眼睛"，舵机是"脖子"，电机是"腿"
 **************************************************/
NewPing sonar(TRIG_PIN, ECHO_PIN, MAX_DISTANCE); // 超声波对象
Servo myservo;  // 舵机对象

/**************************************************
 *            第六部分：状态变量
 * 教学提示：这些是小车的"记忆"
 * 记住自己正在做什么，下一步该做什么
 **************************************************/
boolean goesForward = false;  // 前进状态标志
int distance = 100;           // 存储距离测量值
int avoidState = 0;           // 避障策略状态（0-正常，1-避障中）

/**************************************************
 *            第七部分：setup()函数
 * 教学提示：小车的"出生仪式"
 * 只执行一次，用于初始化所有设置
 **************************************************/
void setup() {
  // 1. 初始化串口调试（小车的"说话能力"）
  Serial.begin(9600);
  Serial.println("========================");
  Serial.println("  智能避障小车启动中...  ");
  Serial.println("========================");
  
  // 2. 设置电机控制引脚为输出模式
  // 教学提示：输出模式 = 发出指令，输入模式 = 接收信息
  pinMode(MOTOR_A_PIN1, OUTPUT);
  pinMode(MOTOR_A_PIN2, OUTPUT);
  pinMode(MOTOR_B_PIN1, OUTPUT);
  pinMode(MOTOR_B_PIN2, OUTPUT);
  
  // 3. 电源稳定等待（给电子元件"热身"时间）
  delay(2000);
  Serial.println("电源稳定检查完成");
  
  // 4. 舵机初始化（根据PDF第11页）
  // 教学提示：舵机就像是小车的"脖子"，可以左右转动
  myservo.attach(9);        // 将舵机连接到引脚9
  myservo.write(90);        // 初始位置：正中间（90度）
  delay(1000);              // 给舵机转动时间
  
  // 5. 电机预测试（检查"腿"是否正常）
  testMotors();
  
  // 6. 初始距离测量（检查"视力"是否正常）
  Serial.println("正在校准超声波传感器...");
  for(int i = 0; i < 3; i++) {
    distance = readPing();
    Serial.print("测量值 ");
    Serial.print(i+1);
    Serial.print(": ");
    Serial.print(distance);
    Serial.println(" cm");
    delay(100);
  }
  
  // 7. 系统就绪提示
  Serial.println("==================================");
  Serial.println("  小车初始化完成！准备出发！  ");
  Serial.println("  使用L9110s驱动板，双TT电机  ");
  Serial.println("==================================");
  Serial.println("");
}

/**************************************************
 *            第八部分：电机测试函数
 * 教学提示：小车的"广播体操"
 * 检查每个动作是否正常
 **************************************************/
void testMotors() {
  Serial.println("开始电机测试...");
  Serial.println("---------------");
  
  // 前转测试（前进5步）
  Serial.println("动作1：前进测试");
  moveForward();
  delay(500);
  moveStop();
  delay(300);
  
  // 后转测试（后退5步）
  Serial.println("动作2：后退测试");
  moveBackward();
  delay(500);
  moveStop();
  delay(300);
  
  // 左转测试（左转90度）
  Serial.println("动作3：左转测试");
  turnLeft();
  delay(500);
  moveStop();
  delay(300);
  
  // 右转测试（右转90度）
  Serial.println("动作4：右转测试");
  turnRight();
  delay(500);
  moveStop();
  delay(300);
  
  Serial.println("---------------");
  Serial.println("电机测试完成！所有动作正常");
  Serial.println("");
}

/**************************************************
 *            第九部分：主循环loop()
 * 教学提示：小车的"大脑思维循环"
 * 不断重复：感知→思考→行动
 **************************************************/
void loop() {
  // 步骤1：感知环境（用"眼睛"看前面有什么）
  distance = readPing();
  
  // 在串口监视器显示信息（调试用）
  Serial.print("前方距离: ");
  Serial.print(distance);
  Serial.print(" cm | 状态: ");
  
  // 步骤2：思考决策（根据看到的东西决定做什么）
  
  // 情况1：紧急情况！太近了！
  if(distance <= CRITICAL_DISTANCE) {
    Serial.println("⚠️ 危险！紧急停止！");
    emergencyStop();
    delay(200);
    moveBackward();    // 先后退
    delay(500);
    avoidObstacle();   // 然后避障
  }
  // 情况2：发现障碍物，需要绕开
  else if(distance <= OBSTACLE_DISTANCE) {
    Serial.println("🚧 检测到障碍物，准备避障");
    moveStop();
    delay(100);
    avoidObstacle();
  }
  // 情况3：前方安全，继续前进
  else {
    Serial.println("✅ 道路通畅，正常前进");
    moveForward();
  }
  
  // 步骤3：短暂休息，准备下一次循环
  delay(50);
}

/**************************************************
 *            第十部分：智能避障决策函数
 * 教学提示：小车的"思考算法"
 * 左右看看，选择最安全的方向
 **************************************************/
void avoidObstacle() {
  Serial.println("");
  Serial.println("🤔 开始避障决策...");
  Serial.println("-------------------");
  
  // 第1步：先停下来，观察环境
  moveStop();
  delay(200);
  
  // 第2步：稍微后退一点，获得更多空间
  Serial.println("动作：后退一点");
  moveBackward();
  delay(300);
  moveStop();
  delay(200);
  
  // 第3步：向右看（转动"脖子"）
  int distanceR = lookRight();
  Serial.print("👀 右侧距离: ");
  Serial.print(distanceR);
  Serial.println(" cm");
  
  // 第4步：向左看
  int distanceL = lookLeft();
  Serial.print("👀 左侧距离: ");
  Serial.print(distanceL);
  Serial.println(" cm");
  
  delay(300);
  
  // 第5步：智能决策（选择最安全的方向）
  Serial.println("💡 决策结果：");
  
  if(distanceR >= SAFE_DISTANCE && distanceL >= SAFE_DISTANCE) {
    // 情况A：两边都安全，选择距离更远的一侧
    if(distanceR > distanceL) {
      Serial.println("✅ 两边都安全，选择右侧（距离更远）");
      turnRight();
    } else {
      Serial.println("✅ 两边都安全，选择左侧（距离更远）");
      turnLeft();
    }
  }
  else if(distanceR >= SAFE_DISTANCE) {
    // 情况B：只有右侧安全
    Serial.println("✅ 右侧安全，向右转");
    turnRight();
  }
  else if(distanceL >= SAFE_DISTANCE) {
    // 情况C：只有左侧安全
    Serial.println("✅ 左侧安全，向左转");
    turnLeft();
  }
  else {
    // 情况D：两边都不安全，180度掉头
    Serial.println("🔄 两边都不安全，执行掉头");
    turnAround();
  }
  
  // 第6步：避障后继续前进
  Serial.println("🚀 避障完成，继续前进");
  delay(300);
  moveForward();
  
  Serial.println("-------------------");
  Serial.println("");
}

/**************************************************
 *            第十一部分：紧急停止函数

void emergencyStop() {
  Serial.println("🛑 紧急制动！");

  analogWrite(MOTOR_A_PIN1, 255);

/**************************************************
 *            第十二部分：180度掉头函数
 * 教学提示：小车的"华丽转身"
 * 原地旋转180度

void turnAround() {
  Serial.println("🌀 执行180度掉头...");
  
  // 原地旋转：一个电机前进，一个后退
  // 左轮前进，右轮后退 → 顺时针旋转
  analogWrite(MOTOR_A_PIN1, TURN_SPEED);  // 左轮前进

  analogWrite(MOTOR_B_PIN1, 0);
  analogWrite(MOTOR_B_PIN2, TURN_SPEED);  // 右轮后退
  
  delay(800);  // 旋转时间，可调整
  
  moveStop();  // 停止
  goesForward = true;  // 更新状态标志
}

int lookRight() {
  myservo.write(30);  // 向右转30度

  int distance = readPing();  // 测量距离

  delay(100);
  myservo.write(90);   // 回到中间
  delay(200);
  return distance;

/**************************************************
 *            第十四部分：距离读取函数
 * 教学提示：小车的"精确测距"
 * 多次测量取平均值，提高准确性
 **************************************************/
int readPing() { 
  int total = 0;          // 总和
  int validReadings = 0;  // 有效读数计数
  
  // 测量3次，取平均值
  for(int i = 0; i < 3; i++) {
    delay(40);  // 每次测量间隔40ms
    int cm = sonar.ping_cm();  // 获取距离值（厘米）
    
    // 过滤无效读数（0或超过最大距离）
    if(cm > 0 && cm < MAX_DISTANCE) {
      total += cm;
      validReadings++;
    }
  }
  
  // 如果有有效读数，计算平均值
  if(validReadings > 0) {
    return total / validReadings;
  } else {
    // 没有有效读数，返回最大距离
    return MAX_DISTANCE;
  }
}

// 停止
void moveStop() {
  // L9110s停止：两个引脚都输出LOW
  analogWrite(MOTOR_A_PIN1, 0);
  analogWrite(MOTOR_A_PIN2, 0);
  analogWrite(MOTOR_B_PIN1, 0);
  analogWrite(MOTOR_B_PIN2, 0);
  goesForward = false;
}

// 前进（带缓启动）
void moveForward() {
  if(!goesForward) {
    goesForward = true;
    
    // 缓启动：从慢到快，减少电流冲击
    Serial.println("缓启动中...");
    for(int i = 80; i <= CRUISE_SPEED; i+=20) {
      analogWrite(MOTOR_A_PIN1, i);
      analogWrite(MOTOR_A_PIN2, 0);
      analogWrite(MOTOR_B_PIN1, i);
      analogWrite(MOTOR_B_PIN2, 0);
      delay(50);
    }
  }
  
  // 保持前进速度
  analogWrite(MOTOR_A_PIN1, CRUISE_SPEED);
  analogWrite(MOTOR_A_PIN2, 0);
  analogWrite(MOTOR_B_PIN1, CRUISE_SPEED);
  analogWrite(MOTOR_B_PIN2, 0);
}

// 后退（带缓启动）
void moveBackward() {
  goesForward = false;
  
  // 缓启动后退
  for(int i = 80; i <= CRUISE_SPEED; i+=20) {
    analogWrite(MOTOR_A_PIN1, 0);
    analogWrite(MOTOR_A_PIN2, i);
    analogWrite(MOTOR_B_PIN1, 0);
    analogWrite(MOTOR_B_PIN2, i);
    delay(50);
  }
  
  // 保持后退速度
  analogWrite(MOTOR_A_PIN1, 0);
  analogWrite(MOTOR_A_PIN2, CRUISE_SPEED);
  analogWrite(MOTOR_B_PIN1, 0);
  analogWrite(MOTOR_B_PIN2, CRUISE_SPEED);
}

// 右转（原地转弯）
void turnRight() {
  Serial.println("↪️  执行右转");
  
  // 右转策略：左轮前进，右轮后退
  analogWrite(MOTOR_A_PIN1, TURN_SPEED);  // 左轮前进
  analogWrite(MOTOR_A_PIN2, 0);
  analogWrite(MOTOR_B_PIN1, 0);
  analogWrite(MOTOR_B_PIN2, TURN_SPEED);  // 右轮后退
  
  delay(400);  // 转弯时间，可调整

}

  Serial.println("↩️  执行左转");

  analogWrite(MOTOR_A_PIN1, 0);
  analogWrite(MOTOR_A_PIN2, TURN_SPEED);  // 左轮后退
  analogWrite(MOTOR_B_PIN1, TURN_SPEED);  // 右轮前进
  analogWrite(MOTOR_B_PIN2, 0);
  
  delay(400);  // 转弯时间，可调整
```
